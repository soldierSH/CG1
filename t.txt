import javax.swing.*;
import java.awt.*;

public class Tela extends JFrame {
    protected PlanoCartesiano planoCartesiano;
    protected JButton botao1;
    protected JButton botao2;
    protected JTextField campo1;
    protected JTextField campo2;
    protected JComboBox<String> opcoes;
    protected JButton botaoDesenhar;

    public Tela() {
        this.initComponents();
    }

    private void initComponents() {
        // Criação dos componentes
        planoCartesiano = new PlanoCartesiano(11, 40);
        botao1 = new JButton("Botão 1");
        botao2 = new JButton("Botão 2");
        campo1 = new JTextField(10);
        campo2 = new JTextField(10);
        String[] listaOpcoes = {"Opção 1", "Opção 2", "Opção 3"};
        opcoes = new JComboBox<>(listaOpcoes);
        botaoDesenhar = new JButton("Desenhar");

        // Configuração da tela
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setTitle("Tela Swing");
        setSize(800, 600);
        setLayout(new BorderLayout());

        // Criação dos painéis
        JPanel painelEsquerdo = new JPanel(new BorderLayout());
        JPanel painelSuperiorEsquerdo = new JPanel();
        JPanel painelBotoes = new JPanel(new GridLayout(2, 2));
        JPanel painelInferior = new JPanel(new FlowLayout(FlowLayout.CENTER));

        // Adição dos componentes aos painéis
        painelSuperiorEsquerdo.add(planoCartesiano);
        painelBotoes.add(botao1);
        painelBotoes.add(campo1);
        painelBotoes.add(botao2);
        painelBotoes.add(campo2);
        painelInferior.add(opcoes);
        painelInferior.add(botaoDesenhar);

        // Adição dos painéis à tela
        painelEsquerdo.add(painelSuperiorEsquerdo, BorderLayout.NORTH);
        painelEsquerdo.add(painelBotoes, BorderLayout.CENTER);
        add(painelEsquerdo, BorderLayout.WEST);
        add(painelInferior, BorderLayout.SOUTH);
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(() -> {
            Tela tela = new Tela();
            tela.setVisible(true);
        });
    }
}
